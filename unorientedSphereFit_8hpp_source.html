85template < class DataPoint, class _WFunctor, typename T>
86FIT_RESULT
87UnorientedSphereFitImpl<DataPoint, _WFunctor, T>::finalize ()
88{
89 PONCA_MULTIARCH_STD_MATH(sqrt);
90 constexpr int Dim = DataPoint::Dim;
91
92 // Compute status
93 if(Base::finalize() != STABLE)
94 return Base::m_eCurrentState;
95 if(Base::getNumNeighbors() < DataPoint::Dim)
96 return Base::m_eCurrentState = UNDEFINED;
97 if (Base::algebraicSphere().isValid())
98 Base::m_eCurrentState = CONFLICT_ERROR_FOUND;
99 else
100 Base::m_eCurrentState = Base::getNumNeighbors() < 2*DataPoint::Dim ? UNSTABLE : STABLE;
101
102 // 1. finalize sphere fitting
103 Scalar invSumW = Scalar(1.) / Base::getWeightSum();
104
105
106 MatrixBB Q;
107 Q.template topLeftCorner<Dim,Dim>().setIdentity();
108 Q.col(Dim).template head<Dim>() = Base::m_sumP * invSumW;
109 Q.row(Dim).template head<Dim>() = Base::m_sumP * invSumW;
110 Q(Dim,Dim) = m_sumDotPP * invSumW;
111 m_matA *= invSumW;
112
113 MatrixBB M = Q.inverse() * m_matA;
114 Eigen::EigenSolver<MatrixBB> eig(M);
115 VectorB eivals = eig.eigenvalues().real();
116 int maxId = 0;
117 eivals.maxCoeff(&maxId);
118
119 VectorB eivec = eig.eigenvectors().col(maxId).real();
120
121 // integrate
122 Base::m_ul = eivec.template head<Dim>();
123 Base::m_uq = Scalar(0.5) * eivec(Dim);
124 Base::m_uc = -invSumW * (Base::m_ul.dot(Base::m_sumP) + m_sumDotPP * Base::m_uq);
125
126 Base::m_isNormalized = false;
127
128 return Base::m_eCurrentState;
129}

155template < class DataPoint, class _WFunctor, typename T, int Type>
156bool
157OrientedSphereDer<DataPoint, _WFunctor, T, Type>::addNeighbor(const DataPoint &_nei)
158{
159 bool bResult = Base::addNeighbor(_nei);
160 if(bResult)
161 {
162 int spaceId = (Type & FitScaleDer) ? 1 : 0;
163
164 ScalarArray w;
165
166 // centered basis
167 VectorType q = _nei.pos() - Base::basisCenter();
168
169 // compute weight
170 if (Type & FitScaleDer)
171 w[0] = Base::m_w.scaledw(q, _nei);
172
173 if (Type & FitSpaceDer){
174 VectorType vw = Base::m_w.spacedw(q, _nei);
175 for(unsigned int i = 0; i < DataPoint::Dim; i++)
176 w[spaceId+i] = vw[i];
177 }
178
179 // increment
180 m_dSumW += w;
181 m_dSumP += q * w;
182 m_dSumN += _nei.normal() * w;
183 m_dSumDotPN += w * _nei.normal().dot(q);
184 m_dSumDotPP += w * q.squaredNorm();
185
186 return true;
187 }
188
189 return false;
190}

193template < class DataPoint, class _WFunctor, typename T, int Type>
194FIT_RESULT
195OrientedSphereDer<DataPoint, _WFunctor, T, Type>::finalize()
196{
197 PONCA_MULTIARCH_STD_MATH(sqrt);
198
199 Base::finalize();
200
201 // Test if base finalize end on a viable case (stable / unstable)
202 if (this->isReady())
203 {
204
205 Scalar invSumW = Scalar(1.)/Base::getWeightSum();
206
207 Scalar nume = Base::m_sumDotPN - invSumW*Base::m_sumP.dot(Base::m_sumN);
208 Scalar deno = Base::m_sumDotPP - invSumW*Base::m_sumP.dot(Base::m_sumP);
209
210 ScalarArray dNume = m_dSumDotPN - invSumW*invSumW * ( Base::getWeightSum() * (
211 Base::m_sumN.transpose() * m_dSumP +
212 Base::m_sumP.transpose() * m_dSumN )
213 - m_dSumW*Base::m_sumP.dot(Base::m_sumN) );
214 ScalarArray dDeno = m_dSumDotPP - invSumW*invSumW*( Scalar(2.)*Base::getWeightSum() * Base::m_sumP.transpose()*m_dSumP
215 - m_dSumW*Base::m_sumP.dot(Base::m_sumP) );
216
217 m_dUq = Scalar(.5) * (deno * dNume - dDeno * nume)/(deno*deno);
218 m_dUl = invSumW*((m_dSumN - Scalar(2.)*(m_dSumP*Base::m_uq+Base::m_sumP*m_dUq)) - Base::m_ul*m_dSumW);
219 m_dUc = -invSumW*( Base::m_sumP.transpose() * m_dUl +
220 Base::m_sumDotPP * m_dUq + Base::m_ul.transpose() * m_dSumP +
221 Base::m_uq*m_dSumDotPP + m_dSumW*Base::m_uc);
222 }
223
224 return Base::m_eCurrentState;
225
226}

229template < class DataPoint, class _WFunctor, typename T, int Type>
230bool
231OrientedSphereDer<DataPoint, _WFunctor, T, Type>::applyPrattNorm()
232{
233 if(Base::isNormalized())
234 return false; //need original parameters without Pratt Normalization
235
236
237 PONCA_MULTIARCH_STD_MATH(sqrt);
238 Scalar pn2 = Base::prattNorm2();
239 Scalar pn = sqrt(pn2);
240
241 ScalarArray dpn2 = dprattNorm2();
242 ScalarArray factor = Scalar(0.5) * dpn2 / pn;
243
244 m_dUc = ( m_dUc * pn - Base::m_uc * factor ) / pn2;
245 m_dUl = ( m_dUl * pn - Base::m_ul * factor ) / pn2;
246 m_dUq = ( m_dUq * pn - Base::m_uq * factor ) / pn2;
247
248 Base::applyPrattNorm();
249 return true;
250}
