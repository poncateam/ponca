101 PONCA_MULTIARCH inline ScalarArray dprattNorm2() const
102 {
103 return Scalar(2.) * Base::m_ul.transpose() * m_dUl

111 return Scalar(2.) * m_dUl.col(_d).dot(Base::m_ul)

113 - Scalar(4.) * m_dUq.col(_d)[0]*Base::m_uc;
114 }
