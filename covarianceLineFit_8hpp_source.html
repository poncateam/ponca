38 template < class DataPoint, class _WFunctor, typename T>
39 FIT_RESULT
40 CovarianceLineFit<DataPoint, _WFunctor, T>::finalize ()
41 {
42  /* handle specific configurations
43  With less than 2 neighbors the fitting is undefined */
44  if(m_sum == Scalar(0.) || Base::m_nbNeighbors < 2)
45  {
46  Base::resetPrimitive();
47  Base::m_eCurrentState = UNDEFINED;
48  return Base::m_eCurrentState;
49  }
50 
51  // Finalize the centroid (still expressed in local basis)
52  m_cog = m_cog/m_sum;
53 
54  // Center the covariance on the centroid
55  m_cov = m_cov/m_sum - m_cog * m_cog.transpose();
56 
57  #ifdef __CUDACC__
58  m_solver.computeDirect(m_cov);
59  #else
60  m_solver.compute(m_cov);
61  #endif
62 
63  Base::m_eCurrentState = ( m_solver.info() == Eigen::Success ? STABLE : UNDEFINED );
64 
65  Base::setLine(m_cog, m_solver.eigenvectors().col(2).normalized());
66 
67  return Base::m_eCurrentState;
68 }
