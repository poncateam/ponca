16 constexpr int clz(unsigned int value)
17 {
18#if defined(__has_builtin) && __has_builtin(__builtin_clz)
19 return __builtin_clz(value);
20#else
21 if (value == 0)
22 {
23 return 0;
24 }
25
26 unsigned int msb_mask = 1 << (sizeof(unsigned int)*8 - 1);
27 int count = 0;
28 for (; (value & msb_mask) == 0; value <<= 1, ++count)
29 {
30 }
31 return count;
32#endif
33 }
