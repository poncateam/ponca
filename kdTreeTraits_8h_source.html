13 constexpr int clz(unsigned int value)
14 {
15#if defined(__has_builtin) && __has_builtin(__builtin_clz)
16 return __builtin_clz(value);
17#else
18 if (value == 0)
19 {
20 return 0;
21 }
22
23 unsigned int msb_mask = 1 << (sizeof(unsigned int)*8 - 1);
24 int count = 0;
25 for (; (value & msb_mask) == 0; value <<= 1, ++count)
26 {
27 }
28 return count;
29#endif
30 }
