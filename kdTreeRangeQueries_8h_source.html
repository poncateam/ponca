29protected:

32public:
33 KdTreeRangeQueryBase(const KdTreeBase<Traits>* kdtree, Scalar radius, typename QueryType::InputType input) :

35
36public:
37 inline Iterator begin(){
38 QueryAccelType::reset();
39 QueryType::reset();
40 Iterator it(this);
41 this->advance(it);
42 return it;
43 }
44 inline Iterator end(){
45 return Iterator(this, QueryAccelType::m_kdtree->point_count());
46 }
47
48protected:
49 inline void advance(Iterator& it){
50 const auto& points = QueryAccelType::m_kdtree->point_data();
51 const auto& indices = QueryAccelType::m_kdtree->index_data();
52 const auto& point = QueryType::getInputPosition(points);
53
54 auto descentDistanceThreshold = [this](){return QueryType::descentDistanceThreshold();};
55 auto skipFunctor = [this](IndexType idx){return QueryType::skipIndexFunctor(idx);};
56 auto processNeighborFunctor = [&it](IndexType idx, IndexType i, Scalar)
57 {
58 it.m_index = idx;
59 it.m_start = i+1;
60 return true;
61 };
62
63 if (points.empty() || indices.empty())
64 throw std::invalid_argument("Empty KdTree");
65
66 for(IndexType i=it.m_start; i<it.m_end; ++i)
67 {
68 IndexType idx = indices[i];
69 if(skipFunctor(idx)) continue;
70
71 Scalar d = (point - points[idx].pos()).squaredNorm();
72 if(d < descentDistanceThreshold())
73 {
74 if( processNeighborFunctor(idx, i, d) ) return;
75 }
76 }
77
78 if (KdTreeQuery<Traits>::search_internal(point,
79 [&it](IndexType start, IndexType end)
80 {
81 it.m_start = start;
82 it.m_end = end;
83 },
84 descentDistanceThreshold,
85 skipFunctor,
86 processNeighborFunctor))
87 it.m_index = static_cast<IndexType>(points.size());
88 }
89};
