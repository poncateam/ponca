52template<class DataPoint>
53template<typename PointUserContainer, typename IndexUserContainer, typename Converter>
54inline void KdTree<DataPoint>::buildWithSampling(const PointUserContainer& points,
55 const IndexUserContainer& sampling,
56 Converter c)
57{
58 this->clear();
59
60 // Copy or convert input samples
61 c( points, m_points );
62
63 m_nodes = NodeContainer();
64 m_nodes.reserve(4 * point_count() / m_min_cell_size);
65 m_nodes.emplace_back();
66 m_nodes.back().leaf = false;
67
68 m_indices = IndexContainer(sampling);//move operator ou std copy
69
70 this->build_rec(0, 0, index_count(), 1);
71
72 PONCA_DEBUG_ASSERT(this->valid());
73}

92template<class DataPoint>
93bool KdTree<DataPoint>::valid() const
94{
95 PONCA_DEBUG_ERROR;
96 return false;
97
98 if (m_points.empty())
99 return m_nodes.empty() && m_indices.empty();
100
101 if(m_nodes.empty() || m_indices.empty())
102 {
103 PONCA_DEBUG_ERROR;
104 return false;
105 }
106
107 if(point_count() < index_count())
108 {
109 PONCA_DEBUG_ERROR;
110 return false;
111 }
112
113 std::vector<bool> b(point_count(), false);
114 for(int idx : m_indices)
115 {
116 if(idx < 0 || point_count() <= idx || b[idx])
117 {
118 PONCA_DEBUG_ERROR;
119 return false;
120 }
121 b[idx] = true;
122 }
123
124 for(int n=0;n<node_count();++n)
125 {
126 const KdTreeNode<Scalar>& node = m_nodes.operator[](n);
127 if(node.leaf)
128 {
129 if(index_count() <= node.start || index_count() < node.start+node.size)
130 {
131 PONCA_DEBUG_ERROR;
132 return false;
133 }
134 }
135 else
136 {
137 if(node.dim < 0 || 2 < node.dim)
138 {
139 PONCA_DEBUG_ERROR;
140 return false;
141 }
142 if(node_count() <= node.firstChildId || node_count() <= node.firstChildId+1u)
143 {
144 PONCA_DEBUG_ERROR;
145 return false;
146 }
147 }
148 }
149
150 return true;
151}

153template<class DataPoint>
154std::string KdTree<DataPoint>::to_string() const
155{
156 if (m_indices.empty()) return "";
157
158 std::stringstream str;
159 str << "indices (" << index_count() << ") :\n";
160 for(int i=0; i<index_count(); ++i)
161 {
162 str << " " << i << ": " << m_indices.operator[](i) << "\n";
163 }
164 str << "nodes (" << node_count() << ") :\n";
165 for(int n=0; n< node_count(); ++n)
166 {
167 const KdTreeNode<Scalar>& node = m_nodes.operator[](n);
168 if(node.leaf)
169 {
170 int end = node.start + node.size;
171 str << " leaf: start=" << node.start << " end=" << end << " (size=" << node.size << ")\n";
172 }
173 else
174 {
175 str << " node: dim=" << node.dim << " split=" << node.splitValue << " child=" << node.firstChildId << "\n";
176 }
177 }
178 return str.str();
179}

193template<class DataPoint>
194void KdTree<DataPoint>::build_rec(int node_id, int start, int end, int level)
195{
196 KdTreeNode<Scalar>& node = m_nodes[node_id];
197 Aabb aabb;
198 for(int i=start; i<end; ++i)
199 aabb.extend(m_points[m_indices[i]].pos());
200
201 int dim;
202 (Scalar(0.5) * (aabb.max() - aabb.min())).maxCoeff(&dim);
203 node.dim = dim;
204 node.splitValue = aabb.center()(dim);
205
206 int midId = this->partition(start, end, dim, node.splitValue);
207 node.firstChildId = m_nodes.size();
208
209 {
210 KdTreeNode<Scalar> n;
211 n.size = 0;
212 m_nodes.push_back(n);
213 m_nodes.push_back(n);
214 }
215 {
216 // left child
217 int childId = m_nodes[node_id].firstChildId;
218 KdTreeNode<Scalar>& child = m_nodes[childId];
219 if(midId-start <= m_min_cell_size || level >= PCA_KDTREE_MAX_DEPTH)
220 {
221 child.leaf = 1;
222 child.start = start;
223 child.size = midId-start;
224 m_leaf_count++;
225 }
226 else
227 {
228 child.leaf = 0;
229 this->build_rec(childId, start, midId, level+1);
230 }
231 }
232 {
233 // right child
234 int childId = m_nodes[node_id].firstChildId+1;
235 KdTreeNode<Scalar>& child = m_nodes[childId];
236 if(end-midId <= m_min_cell_size || level >= PCA_KDTREE_MAX_DEPTH)
237 {
238 child.leaf = 1;
239 child.start = midId;
240 child.size = end-midId;
241 m_leaf_count++;
242 }
243 else
244 {
245 child.leaf = 0;
246 this->build_rec(childId, midId, end, level+1);
247 }
248 }
249}

251template<class DataPoint>
252int KdTree<DataPoint>::partition(int start, int end, int dim, Scalar value)
253{
254 const auto& points = m_points;
255 auto& indices = m_indices;
256
257 auto it = std::partition(indices.begin()+start, indices.begin()+end, [&](int i)
258 {
259 return points[i].pos()[dim] < value;
260 });
261
262 auto distance = std::distance(m_indices.begin(), it);
263
264 return static_cast<int>(distance);
265}
