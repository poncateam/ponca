108 PONCA_MULTIARCH inline void changeBasis(const VectorType& newbasis)
109 {
110 VectorType diff = Base::m_w.basisCenter() - newbasis;
111 Base::m_w.init( newbasis );
112 m_uc = m_uc - m_ul.dot(diff) + m_uq * diff.dot(diff);
113 m_ul = m_ul - Scalar(2.)*m_uq*diff;
114 //m_uq is not changed
115 m_isNormalized = false;
116 applyPrattNorm();
117 }

136 PONCA_MULTIARCH inline bool applyPrattNorm()
137 {
138 if (! m_isNormalized)
139 {
140 Scalar pn = prattNorm();
141 m_uc /= pn;
142 m_ul *= Scalar(1.)/pn;
143 m_uq /= pn;
144
145 m_isNormalized = true;
146 }
147 return true;
148 }

154 PONCA_MULTIARCH inline Scalar radius() const
155 {
156 if(isPlane())
157 {
158 //return infinity (non-sense value)
159#ifdef __CUDACC__

161 return Scalar(1.)/inf;
162#else
163 return std::numeric_limits<Scalar>::infinity();

167 PONCA_MULTIARCH_STD_MATH(sqrt);
168 Scalar b = Scalar(1.)/m_uq;
169 return Scalar(sqrt( ((Scalar(-0.5)*b)*m_ul).squaredNorm() - m_uc*b ));
170 }

176 PONCA_MULTIARCH inline VectorType center() const

185#endif
186 }
187
188 Scalar b = Scalar(1.)/m_uq;
189 return (Scalar(-0.5)*b)*m_ul + Base::m_w.basisCenter();
190 }
