30 typename DistWeightFunc<DataPoint, WeightKernel>::MatrixType
31 DistWeightFunc<DataPoint, WeightKernel>::spaced2w( const VectorType& _q,
32  const DataPoint&) const
33 {
34  MatrixType result = MatrixType::Zero();
35  Scalar d = _q.norm();
36  if (d <= m_t && d != Scalar(0.))
37  {
38  Scalar der = m_wk.df(d/m_t);
39  result = _q*_q.transpose()/d*(m_wk.ddf(d/m_t)/m_t - der/d);
40  result.diagonal().array() += der;
41  result *= Scalar(1.)/(m_t*d);
42  }
43  return result;
44 }
