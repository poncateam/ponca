105 PONCA_MULTIARCH inline void advance(Iterator& iterator)
106 {
107 const auto& points = m_graph->points();
108 const auto& point = points[QueryType::input()].pos();
109
110 if (!(iterator != end()))
111 return;
112
113 if (m_stack.empty())
114 {
115 iterator = end();
116 }
117 else
118 {
119 int idx_current = m_stack.top();
120 m_stack.pop();
121
122 PONCA_DEBUG_ASSERT((point - points[idx_current].pos()).squaredNorm() < QueryType::squaredRadius());
123
124 iterator.m_index = idx_current;
125
126 for (int idx_nei : m_graph->kNearestNeighbors(idx_current))
127 {
128 PONCA_DEBUG_ASSERT(idx_nei >= 0);
129 Scalar d = (point - points[idx_nei].pos()).squaredNorm();
130 Scalar th = QueryType::descentDistanceThreshold();
131
132 if ((point - points[idx_nei].pos()).squaredNorm() < QueryType::descentDistanceThreshold() &&
133 m_flag.insert(idx_nei))
134 {
135 m_stack.push(idx_nei);
136 }
137 }
138 if (iterator.m_index == QueryType::input())
139 advance(iterator); // query is not included in returned set
140 }
141 }
