202 template <typename KdTreeTraits>
203 PONCA_MULTIARCH_HOST inline KnnGraphBase(const KdTreeBase<KdTreeTraits>& _kdtree, const int _k = 6) : Base(_k)
204 // : Base({std::min(_k, _kdtree.sampleCount() - 1)})
205 // : Base(typename Base::Buffers(std::min(_k, _kdtree.sampleCount() - 1)))
206 {
207 Base::m_bufs.points_size = _kdtree.pointCount();
208 Base::m_bufs.points = std::move(_kdtree.points());
209 static_assert(std::is_same_v<typename Traits::DataPoint, typename KdTreeTraits::DataPoint>,
210 "KdTreeTraits::DataPoint is not equal to Traits::DataPoint");
211 static_assert(std::is_same_v<typename Traits::PointContainer, typename KdTreeTraits::PointContainer>,
212 "KdTreeTraits::PointContainer is not equal to Traits::PointContainer");
213 static_assert(std::is_same_v<typename Traits::IndexContainer, typename KdTreeTraits::IndexContainer>,
214 "KdTreeTraits::IndexContainer is not equal to Traits::IndexContainer");
215
216 // We need to account for the entire point set, irrespectively of the sampling. This is because the kdtree
217 // (kNearestNeighbors) return ids of the entire point set, not it sub-sampled list of ids.
218 // \fixme Update API to properly handle kdtree subsampling
219 const int cloudSize = _kdtree.pointCount();
220 {
221 const int samplesSize = _kdtree.sampleCount();
222 PONCA_ASSERT(cloudSize == samplesSize);
223 }
224
225 Base::m_bufs.indices_size = cloudSize * Base::m_bufs.k;
226 Base::m_bufs.indices.resize(Base::m_bufs.indices_size, -1);
227
228#pragma omp parallel for shared(_kdtree, cloudSize) default(none)
229 for (int i = 0; i < cloudSize; ++i)
230 {
231 int j = 0;
232 for (auto n : _kdtree.kNearestNeighbors(typename KdTreeTraits::IndexType(i),
233 typename KdTreeTraits::IndexType(Base::m_bufs.k)))
234 {
235 Base::m_bufs.indices[i * Base::m_bufs.k + j] = n;
236 ++j;
237 }
238 }
239 }
240 };
