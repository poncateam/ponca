133 PONCA_MULTIARCH inline ScalarArray dprattNorm2() const
134 {
135 return Scalar(2.) * Base::m_ul.transpose() * m_dUl
136 - Scalar(4.) * Base::m_uq * m_dUc
137 - Scalar(4.) * Base::m_uc * m_dUq;
138 }
139
141 PONCA_MULTIARCH inline Scalar dprattNorm2(unsigned int _d) const
142 {
143 return Scalar(2.) * m_dUl.col(_d).dot(Base::m_ul)
144 - Scalar(4.) * m_dUc.col(_d)[0]*Base::m_uq
145 - Scalar(4.) * m_dUq.col(_d)[0]*Base::m_uc;
146 }
147
149 PONCA_MULTIARCH inline Scalar dprattNorm(unsigned int _d) const
150 {
151 PONCA_MULTIARCH_STD_MATH(sqrt);
152 return sqrt(dprattNorm2(_d));
153 }

157 {
158 PONCA_MULTIARCH_STD_MATH(sqrt);
