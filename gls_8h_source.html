98 PONCA_MULTIARCH inline Scalar compareTo (const GLSParam<DataPoint, _WFunctor, T>& _other,
99 bool _useFitness = true) const
100 {
101 Scalar nTau = this->tau_normalized() - _other.tau_normalized();
102 Scalar nKappa = this->kappa_normalized() - _other.kappa_normalized();
