479 template<typename PointUserContainer, typename PointConverter>
480 PONCA_MULTIARCH_HOST inline void build(PointUserContainer&& points, PointConverter c);
481
483 struct DefaultConverter
484 {
485 template <typename Input>
486 PONCA_MULTIARCH_HOST inline void operator()(Input&& i, PointContainer& o)
487 {
488 using InputContainer = std::remove_reference_t<Input>;
489 if constexpr (std::is_same_v<InputContainer, PointContainer> && std::is_copy_assignable_v<DataPoint>)
490 o = std::forward<Input>(i); // Either move or copy
491 else
492 std::transform(i.cbegin(), i.cend(), std::back_inserter(o),
493 [](const typename InputContainer::value_type &p) -> DataPoint { return DataPoint(p); });
494 }
495 };
496
500 template<typename PointUserContainer>
501 PONCA_MULTIARCH_HOST inline void build(PointUserContainer&& points)
502 {
503 build(std::forward<PointUserContainer>(points), DefaultConverter());
504 }
505
506 // Internal ----------------------------------------------------------------
507protected:
515 template<typename PointUserContainer, typename IndexUserContainer, typename PointConverter>
516 PONCA_MULTIARCH_HOST inline void buildWithSampling(
517 PointUserContainer&& points,
518 IndexUserContainer&& sampling,
519 PointConverter c
520 );
521
527 template<typename PointUserContainer, typename IndexUserContainer>
528 PONCA_MULTIARCH_HOST inline void buildWithSampling(PointUserContainer&& points, IndexUserContainer&& sampling)
529 {
530 buildWithSampling(std::forward<PointUserContainer>(points), std::forward<IndexUserContainer>(sampling), DefaultConverter());
531 }
532
533private:
534 PONCA_MULTIARCH_HOST inline void buildRec(NodeIndexType node_id, IndexType start, IndexType end, int level);
535 PONCA_MULTIARCH_HOST [[nodiscard]] inline IndexType partition(IndexType start, IndexType end, int dim, Scalar value);
536
537};
