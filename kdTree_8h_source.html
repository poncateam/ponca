133 struct DefaultConverter
134 {
135 template <typename Input>
136 inline void operator()(Input&& i, PointContainer& o)
137 {
138 using InputContainer = typename std::remove_reference<Input>::type;
139 if constexpr (std::is_same<InputContainer, PointContainer>::value)
140 o = std::forward<Input>(i); // Either move or copy
141 else
142 std::transform(i.cbegin(), i.cend(), std::back_inserter(o),
143 [](const typename InputContainer::value_type &p) -> DataPoint { return DataPoint(p); });
144 }
145 };
