126 struct DefaultConverter
127 {
128 template <typename Input>
129 inline void operator()(Input&& i, PointContainer& o)
130 {
131 using InputContainer = typename std::remove_reference<Input>::type;
132 if constexpr (std::is_same<InputContainer, PointContainer>::value)
133 o = std::forward<Input>(i); // Either move or copy
134 else
135 std::transform(i.cbegin(), i.cend(), std::back_inserter(o),
136 [](const typename InputContainer::value_type &p) -> DataPoint { return DataPoint(p); });
137 }
138 };
