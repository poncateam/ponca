130template < class DataPoint, class _WFunctor, int Type, typename T>
131class PrimitiveDer : public T
132{
133private:
134 typedef T Base;
136protected:
137 enum {
138 Check = Base::PROVIDES_PRIMITIVE_BASE,
139 PROVIDES_PRIMITIVE_DERIVATIVE
140 };
141
142protected:
143 static constexpr int NbDerivatives = ((Type & FitScaleDer) ? 1 : 0 ) + ((Type & FitSpaceDer) ? DataPoint::Dim : 0);
144 static constexpr int DerStorageOrder = (Type & FitSpaceDer) ? Eigen::RowMajor : Eigen::ColMajor;
145
146public:
147 using Scalar = typename Base::Scalar;
148 using VectorType = typename Base::VectorType;
149 using MatrixType = typename DataPoint::MatrixType;
150 using WFunctor = typename Base::WFunctor;
153 typedef Eigen::Matrix<Scalar, DataPoint::Dim, NbDerivatives, DerStorageOrder> VectorArray;
154
156 typedef Eigen::Matrix<Scalar, 1, NbDerivatives> ScalarArray;
157
158protected:
159 // computation data
160 ScalarArray m_dSumW;
162public:
163
164 /************************************************************************/
165 /* Initialization */
166 /************************************************************************/
168 PONCA_MULTIARCH inline void init(const VectorType &_evalPos)
169 { Base::init(_evalPos); m_dSumW.setZero(); }
170
171 /************************************************************************/
172 /* Processing */
173 /************************************************************************/
176 PONCA_MULTIARCH inline bool addLocalNeighbor(Scalar w,
177 const VectorType &localQ,
178 const DataPoint &attributes,
179 ScalarArray &dw)
180 {
181 if( Base::addLocalNeighbor(w, localQ, attributes) ) {
182 int spaceId = (Type & FitScaleDer) ? 1 : 0;
183 // compute weight
184 if (Type & FitScaleDer)
185 dw[0] = Base::m_w.scaledw(attributes.pos(), attributes);
186
187 if (Type & FitSpaceDer)
188 dw.template segment<int(DataPoint::Dim)>(spaceId) = -Base::m_w.spacedw(attributes.pos(), attributes).transpose();
189
190 m_dSumW += dw;
191 return true;
192 }
193 return false;
194 }
195
196 /**************************************************************************/
197 /* Use results */
198 /**************************************************************************/
200 PONCA_MULTIARCH inline constexpr bool isScaleDer() const {return bool(Type & FitScaleDer);}
202 PONCA_MULTIARCH inline constexpr bool isSpaceDer() const {return bool(Type & FitSpaceDer);}
204 PONCA_MULTIARCH inline constexpr unsigned int derDimension() const { return NbDerivatives;}
205
206};
