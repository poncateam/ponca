133template < class DataPoint, class _WFunctor, int Type, typename T>
134class PrimitiveDer : public T
135{
136private:
137 typedef T Base;
139protected:
140 enum {
141 Check = Base::PROVIDES_PRIMITIVE_BASE,
142 PROVIDES_PRIMITIVE_DERIVATIVE
143 };
144
145protected:
146 static constexpr int NbDerivatives = ((Type & FitScaleDer) ? 1 : 0 ) + ((Type & FitSpaceDer) ? DataPoint::Dim : 0);
147 static constexpr int DerStorageOrder = (Type & FitSpaceDer) ? Eigen::RowMajor : Eigen::ColMajor;
148
149public:
150 using Scalar = typename Base::Scalar;
151 using VectorType = typename Base::VectorType;
152 using MatrixType = typename DataPoint::MatrixType;
153 using WFunctor = typename Base::WFunctor;
156 typedef Eigen::Matrix<Scalar, DataPoint::Dim, NbDerivatives, DerStorageOrder> VectorArray;
157
159 typedef Eigen::Matrix<Scalar, 1, NbDerivatives> ScalarArray;
160
161protected:
162 // computation data
163 ScalarArray m_dSumW;
165public:
166
167 /************************************************************************/
168 /* Initialization */
169 /************************************************************************/
171 PONCA_MULTIARCH inline void init(const VectorType &_evalPos)
172 { Base::init(_evalPos); m_dSumW.setZero(); }
173
174 /************************************************************************/
175 /* Processing */
176 /************************************************************************/
179 PONCA_MULTIARCH inline bool addLocalNeighbor(Scalar w,
180 const VectorType &localQ,
181 const DataPoint &attributes,
182 ScalarArray &dw)
183 {
184 if( Base::addLocalNeighbor(w, localQ, attributes) ) {
185 int spaceId = (Type & FitScaleDer) ? 1 : 0;
186 // compute weight
187 if (Type & FitScaleDer)
188 dw[0] = Base::m_w.scaledw(attributes.pos(), attributes);
189
190 if (Type & FitSpaceDer)
191 dw.template segment<int(DataPoint::Dim)>(spaceId) = -Base::m_w.spacedw(attributes.pos(), attributes).transpose();
192
193 m_dSumW += dw;
194 return true;
195 }
196 return false;
197 }
198
199 /**************************************************************************/
200 /* Use results */
201 /**************************************************************************/
203 PONCA_MULTIARCH inline constexpr bool isScaleDer() const {return bool(Type & FitScaleDer);}
205 PONCA_MULTIARCH inline constexpr bool isSpaceDer() const {return bool(Type & FitSpaceDer);}
207 PONCA_MULTIARCH inline constexpr unsigned int derDimension() const { return NbDerivatives;}
208
209};
