159 template <typename IteratorBegin, typename IteratorEnd>
160 PONCA_MULTIARCH inline FIT_RESULT compute(const IteratorBegin& begin, const IteratorEnd& end){
161 Base::init();
162 FIT_RESULT res = UNDEFINED;
163
164 do {
165 derived().startNewPass();
166 for (auto it = begin; it != end; ++it){
167 derived().addNeighbor(*it);
168 }
169 res = Base::finalize();
170 } while ( res == NEED_OTHER_PASS );
171
172 return res;
173 }

192 res = Base::finalize();
