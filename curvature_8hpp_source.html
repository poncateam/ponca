23 template < class DataPoint, class _WFunctor, typename T>
24 FIT_RESULT CurvatureEstimator<DataPoint, _WFunctor, T>::computeCurvature(bool useNormal)
25 {
26  PONCA_MULTIARCH_STD_MATH(abs);
27 
28  // Get the object space Weingarten map dN
29  MatrixType dN = Base::dNormal().template middleCols<DataPoint::Dim>(Base::isScaleDer() ? 1: 0);
30 
31  // Compute tangent-space basis
32  Mat32 B = tangentPlane(useNormal);
33 
34  // Compute the 2x2 matrix representing the shape operator by transforming dN to the basis B.
35  // Recall that dN is a bilinear form, it thus transforms as follows:
36  Mat22 S = B.transpose() * dN * B;
37 
38  // Recall that at this stage, the shape operator represented by S describes the normal curvature K_n(u) in the direction u \in R^2 as follows:
39  // K_n(u) = u^T S u
40  // The principal curvatures are fully defined by the values and the directions of the extrema of K_n.
41  //
42  // If the normal field N(x) comes from the gradient of a scalar field, then N(x) is curl-free, and dN and S are symmetric matrices.
43  // In this case, the extrema of the previous quadratic form are directly obtained by the the eigenvalue decomposition of S.
44  // However, if N(x) is only an approximation of the normal field of a surface, then N(x) is not necessarily curl-free, and in this case S is not symmetric.
45  // In this case, we first have to find an equivalent symmetric matrix S' such that:
46  // K_n(u) = u^T S' u,
47  // for any u \in R^2.
48  // It is easy to see that such a S' is simply obtained as:
49  // S' = (S + S^T)/2
50 
51  S(0,1) = S(1,0) = (S(0,1) + S(1,0))/Scalar(2);
52  Eigen::SelfAdjointEigenSolver<Mat22> eig2;
53  eig2.computeDirect(S);
54 
55  if (eig2.info() != Eigen::Success){
56  return UNDEFINED;
57  }
58 
59  m_k1 = eig2.eigenvalues()(0);
60  m_k2 = eig2.eigenvalues()(1);
61 
62  m_v1 = B * eig2.eigenvectors().col(0);
63  m_v2 = B * eig2.eigenvectors().col(1);
64 
65  if(abs(m_k1)<abs(m_k2))
66  {
67 #ifdef __CUDACC__
68  Scalar tmpk = m_k1;
69  m_k1 = m_k2;
70  m_k2 = tmpk;
71  VectorType tmpv = m_v1;
72  m_v1 = m_v2;
73  m_v2 = tmpv;
74 #else
75  std::swap(m_k1, m_k2);
76  std::swap(m_v1, m_v2);
77 #endif
78  }
79 
80  return this->getCurrentState();
81 }

83 template < class DataPoint, class _WFunctor, typename T>
84 typename CurvatureEstimator<DataPoint, _WFunctor, T>::Mat32
85 CurvatureEstimator<DataPoint, _WFunctor, T>::tangentPlane(bool useNormal) const
86 {
87  typedef typename VectorType::Index Index;
88 
89  PONCA_MULTIARCH_STD_MATH(sqrt);
90  PONCA_MULTIARCH_STD_MATH(abs);
91 
92  Mat32 B;
93  Index i0=Index(-1), i1=Index(-1), i2=Index(-1);
94 
95  // Two choices to compute a basis of the tangent plane
96  if(useNormal)
97  {
98  // Use a vector orthogonal to the surface (the gradient) and compute an
99  // orthonormal basis from it
100  VectorType n = Base::primitiveGradient();
101  n.array().abs().minCoeff(&i0); // i0: dimension where n extends the least
102  i1 = (i0+1)%3;
103  i2 = (i0+2)%3;
104 
105  B.col(0)[i0] = 0;
106  B.col(0)[i1] = n[i2];
107  B.col(0)[i2] = -n[i1];
108 
109  B.col(0).normalize();
110  B.col(1) = B.col(0).cross(n);
111  }
112  else
113  {
114  // Use the spatial derivative of the normal. This option leads to NaN
115  // values if dN is null (like in the case of a perfect plane)
116 
117  // Get the object space Weingarten map dN
118  MatrixType dN = Base::dNormal().template middleCols<DataPoint::Dim>(Base::isScaleDer() ? 1: 0);
119 
120  // Compute tangent-space basis from dN
121  // 1 - pick the column with maximal norm as the first tangent vector,
122  Scalar sqNorm = dN.colwise().squaredNorm().maxCoeff(&i0);
123  B.col(0) = dN.col(i0) / sqrt(sqNorm);
124  // 2 - orthogonalize the other column vectors, and pick the most reliable one
125  i1 = (i0+1)%3;
126  i2 = (i0+2)%3;
127  VectorType v1 = dN.col(i1) - B.col(0).dot(dN.col(i1)) * B.col(0);
128  VectorType v2 = dN.col(i2) - B.col(0).dot(dN.col(i2)) * B.col(0);
129  Scalar v1norm2 = v1.squaredNorm();
130  Scalar v2norm2 = v2.squaredNorm();
131  if(v1norm2 > v2norm2) B.col(1) = v1 / sqrt(v1norm2);
132  else B.col(1) = v2 / sqrt(v2norm2);
133  }
134 
135  return B;
136 }
