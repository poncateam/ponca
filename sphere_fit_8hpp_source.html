22 bool
23 SphereFit<DataPoint, _WFunctor, T>::addNeighbor(const DataPoint& _nei)
24 {
25  // centered basis
26  VectorType q = _nei.pos() - Base::basisCenter();
27 
28  // compute weight
29  Scalar w = m_w.w(q, _nei);
30 
31  if (w > Scalar(0.))
32  {
33  VectorA a;
34 #ifdef __CUDACC__
35  a(0) = 1;
36  a.template segment<DataPoint::Dim>(1) = q;
37  a(DataPoint::Dim+1) = q.squaredNorm();
38 #else
39  a << 1, q, q.squaredNorm();
40 #endif
41  m_matA += w * a * a.transpose();
42  m_sumW += w;
43 
44  ++(Base::m_nbNeighbors);
45  return true;
46  }
47

52 template < class DataPoint, class _WFunctor, typename T>
53 FIT_RESULT
54 SphereFit<DataPoint, _WFunctor, T>::finalize ()
55 {
56  // handle specific configurations
57  // With less than 3 neighbors the fitting is undefined
58  if(m_sumW == Scalar(0.) || Base::m_nbNeighbors < 3)
59  {
60  Base::m_ul.setZero();
61  Base::m_uc = Scalar(0.);
62  Base::m_uq = Scalar(0.);
63  Base::m_isNormalized = false;
64  Base::m_eCurrentState = UNDEFINED;
65  return Base::m_eCurrentState;
66  }
67 
68  MatrixA matC;
69  matC.setIdentity();
70  matC.template topRightCorner<1,1>() << -2;
71  matC.template bottomLeftCorner<1,1>() << -2;
72  matC.template topLeftCorner<1,1>() << 0;
73  matC.template bottomRightCorner<1,1>() << 0;
74 
75  MatrixA invCpratt;
76  invCpratt.setIdentity();
77  invCpratt.template topRightCorner<1,1>() << -0.5;
78  invCpratt.template bottomLeftCorner<1,1>() << -0.5;
79  invCpratt.template topLeftCorner<1,1>() << 0;
80  invCpratt.template bottomRightCorner<1,1>() << 0;
81 
82  MatrixA M = invCpratt * m_matA;
83  // go to positive semi-definite matrix to be compatible with
84  // SelfAdjointEigenSolver requirements
85  // Note: This does not affect the eigen vectors order
86  Eigen::SelfAdjointEigenSolver<MatrixA> solver;
87 #ifdef __CUDACC__
88  solver.computeDirect(M.transpose() * M);
89 #else
90  solver.compute(M.transpose() * M);
91 #endif
92  VectorA eivals = solver.eigenvalues().real();
93  int minId = -1;
94  for(int i=0 ; i<DataPoint::Dim+2 ; ++i)
95  {
96  Scalar ev = eivals(i);
97  if((ev>0) && (minId==-1 || ev<eivals(minId)))
98  minId = i;
99  }
100 
101  //mLambda = eivals(minId);
102  VectorA vecU = solver.eigenvectors().col(minId).real();
103  Base::m_uq = vecU[1+DataPoint::Dim];
104  Base::m_ul = vecU.template segment<DataPoint::Dim>(1);
105  Base::m_uc = vecU[0];
106 
107  Base::m_isNormalized = false;
108 
109  if(Base::m_nbNeighbors < 6)
110  {
111  Base::m_eCurrentState = UNSTABLE;
112  }
113  else
114  {
115  Base::m_eCurrentState = STABLE;
116  }
117 
118  return Base::m_eCurrentState;
119 }
